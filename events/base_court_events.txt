# Your young wife takes a passion for romantic poetry. She keeps reading any sort of romantic poetry she can get hold of.

	is_triggered_only = yes # on_yearly_pulse, random_events

	only_playable = yes
	only_men = yes
	is_married = yes

	prisoner = no

	trigger = {
		spouse = {
			age < 21
			fertility >= 0.15
			is_ruler = no

			prisoner = no
		}

		trigger_if = {
			limit = { has_character_flag = event_73001 }

			had_character_flag = {
				flag = event_73001
				years >= 5
			}
		}
	}

	weight_multiplier = {
		factor = 1

		mult_modifier = {
			factor = 4.0
			spouse = { is_pregnant = yes }
		}

		mult_modifier = {
			factor = 2.0
			spouse = { trait = poet }
		}

		mult_modifier = {
			factor = 1.5
			trait = gregarious
		}
	}

	immediate = {
		set_character_flag = event_73001
	}

	option = {
		name = EVTOPTA73001 # Help her, by getting her more to read.

		trigger = { wealth >= 25 }

		wealth = -25

		spouse = {

			}

			opinion = {
				name = spouse_encouraged

				years = 1
			}
		}
	}

	option = {
		name = EVTOPTB73001 # That is her interest, not mine.

		piety = 5

		spouse = {
			show_scope_change = no

			fertility = -0.1

			opinion = {
				name = spouse_ignores_me
				who = ROOT
				years = 1
			}
		}
	}

	option = {
		name = {
			text = EVTOPTC73001 # These are books created by satan!
			trigger = { religion_group = christian }
		}
		name = {
			text = EVTOPTD73001 # These are books created by Shaytan!
			trigger = { religion_group = muslim }
		}

		trigger = {
			OR = {
				religion_group = christian
