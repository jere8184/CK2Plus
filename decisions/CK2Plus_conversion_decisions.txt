	convert_to_roman = {
		is_high_prio = yes
		ai_check_interval = 12

		only_independent = yes

		potential = {
			has_landed_title = e_roman_empire

			NOR = {
				culture = roman
				has_dlc = "Holy Fury" # Use Roman Renaissance chain instead
			}
		}

		allow = {
			ruled_years >= 5
		}

		effect = {
			culture = roman

			any_courtier = {
				show_scope_change = no

				limit = {
					dynasty = ROOT
					is_ruler = no
				}

				culture = roman
			}

			capital_scope = {
				show_scope_change = no

				culture = roman

				add_province_modifier = {
					name = recent_culture_change
					duration = -1
				}
			}
		}

		ai_will_do = {
			factor = 1

			mult_modifier = {
				factor = 1.1

				num_culture_realm_provs = {
					value >= 2
					culture = roman
				}
			}

			mult_modifier = {
				factor = 1.1

				num_culture_realm_provs = {
					value >= 4
					culture = roman
				}
			}

			mult_modifier = {
				factor = 1.1

				num_culture_realm_provs = {
					value >= 6
					culture = roman
				}
			}

			mult_modifier = {
				factor = 1.1

				num_culture_realm_provs = {
					value >= 8
					culture = roman
				}
			}

			mult_modifier = {
				factor = 1.1

				num_culture_realm_provs = {
					value >= 10
					culture = roman
				}
			}
		}
	}

	gone_pagan_zoroastrian = {
		is_high_prio = yes
		ai_check_interval = 12

		potential = {
			zoroastrian_culture_trigger = yes
			is_adult = yes
			is_heretic = no
			controls_religion = no
			NOT = { rightful_religious_head = ROOT }

			trigger_if = {
				limit = { is_ruler = yes }
				capital_holding = { NOT = { holding_type = temple } }
			}

			NOR = {
				religion = zoroastrian
				religion = jewish
				religion_group = pagan_group
				has_character_modifier = voice_of_jesus
				has_character_modifier = voice_of_satan
				has_character_flag = gone_pagan
			}

			trigger_if = {
				limit = { ai = no }

				has_character_flag = seeks_pagan_knowledge
				has_dlc = "The Old Gods"
			}
		}

		allow = {
			OR = {
				custom_tooltip = {
					text = RECEIVED_INSTRUCTION_MYSTIC
					has_character_flag = may_go_pagan
				}

				any_liege = { religion = zoroastrian }
			}

			OR = {
				trait = mystic
				trait = cynical
			}

			trait = sympathy_zoroastrianism
			NOT = { trait = zealous }
			has_secret_religion = no
			war = no
			in_command_trigger = no
			is_not_in_regency_trigger = yes
			is_inaccessible_trigger = no
		}

		effect = {
			set_character_flag = gone_pagan

			character_event = {
				id = GoPagan.100
				tooltip = CONSIDER_RELIGION_CHANGE
			}
		}

		ai_will_do = {
			factor = 1

			trigger = {
				OR = {
					NOT = { religion_authority >= 0.10 }

					AND = {
						any_demesne_province = {
							religion_group = zoroastrian_group
							owned_by = PREV
						}

						NOT = { religion_authority = 0.20 }
					}

					AND = {
						trait = lunatic
						NOT = { religion_authority = 0.40 }
					}
				}

				NOT = { religion_authority >= 0.5 }
			}
		}
	}
